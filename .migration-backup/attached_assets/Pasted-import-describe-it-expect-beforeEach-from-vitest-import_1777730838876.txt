import { describe, it, expect, beforeEach } from 'vitest';
import { prisma, TEST_DATA } from './setup';
import { createUserContext, createTestDepartment, makeAuthenticatedRequest } from './helpers';
import { UserRole } from '@prisma/client';

describe('Department Scope - HOD & Faculty Isolation', () => {
  let hodPhysics: any;
  let hodMath: any;
  let facultyPhysics: any;
  let facultyMath: any;
  let physicsClass: any;
  let mathClass: any;

  beforeEach(async () => {
    // Create departments
    await createTestDepartment(TEST_DATA.departments.physics, TEST_DATA.schools.schoolA, 'Physics');
    await createTestDepartment(TEST_DATA.departments.mathematics, TEST_DATA.schools.schoolA, 'Mathematics');

    // Create HODs
    hodPhysics = await createUserContext(
      TEST_DATA.users.hodPhysics,
      'hod.physics@school.test',
      'HOD Physics',
      UserRole.ADMIN, // Assume HOD is ADMIN role in your schema
      TEST_DATA.schools.schoolA,
      TEST_DATA.departments.physics
    );

    hodMath = await createUserContext(
      TEST_DATA.users.hodMath,
      'hod.math@school.test',
      'HOD Math',
      UserRole.ADMIN,
      TEST_DATA.schools.schoolA,
      TEST_DATA.departments.mathematics
    );

    // Create Faculty
    facultyPhysics = await createUserContext(
      TEST_DATA.users.facultyPhysics,
      'faculty.physics@school.test',
      'Faculty Physics',
      UserRole.FACULTY,
      TEST_DATA.schools.schoolA,
      TEST_DATA.departments.physics
    );

    facultyMath = await createUserContext(
      TEST_DATA.users.facultyMath,
      'faculty.math@school.test',
      'Faculty Math',
      UserRole.FACULTY,
      TEST_DATA.schools.schoolA,
      TEST_DATA.departments.mathematics
    );

    // Create classes
    physicsClass = await prisma.class.upsert({
      where: { id: 'class_physics_10a' },
      update: {},
      create: {
        id: 'class_physics_10a',
        schoolId: TEST_DATA.schools.schoolA,
        name: 'Physics 10A',
        grade: '10',
        section: 'A',
        subject: 'Physics',
        facultyId: `faculty_${TEST_DATA.users.facultyPhysics}`,
      },
    });

    mathClass = await prisma.class.upsert({
      where: { id: 'class_math_10b' },
      update: {},
      create: {
        id: 'class_math_10b',
        schoolId: TEST_DATA.schools.schoolA,
        name: 'Math 10B',
        grade: '10',
        section: 'B',
        subject: 'Mathematics',
        facultyId: `faculty_${TEST_DATA.users.facultyMath}`,
      },
    });
  });

  it('HOD Physics should access Physics classes', async () => {
    const response = await makeAuthenticatedRequest('GET', `/api/classes/${physicsClass.id}`, hodPhysics.jwt);

    expect(response.status).toBe(200);
    expect(response.data.class?.subject).toBe('Physics');
  });

  it('HOD Physics should NOT access Math classes', async () => {
    const response = await makeAuthenticatedRequest('GET', `/api/classes/${mathClass.id}`, hodPhysics.jwt);

    // Should fail
    expect([403, 404, 200]).toContain(response.status);
    if (response.status === 200) {
      expect(response.data.class).toBeNull();
    }
  });

  it('HOD Math should access Math classes', async () => {
    const response = await makeAuthenticatedRequest('GET', `/api/classes/${mathClass.id}`, hodMath.jwt);

    expect(response.status).toBe(200);
    expect(response.data.class?.subject).toBe('Mathematics');
  });

  it('HOD Math should NOT access Physics classes', async () => {
    const response = await makeAuthenticatedRequest('GET', `/api/classes/${physicsClass.id}`, hodMath.jwt);

    expect([403, 404, 200]).toContain(response.status);
    if (response.status === 200) {
      expect(response.data.class).toBeNull();
    }
  });

  it('Faculty Physics should see own department classes', async () => {
    const response = await makeAuthenticatedRequest('GET', `/api/classes/${physicsClass.id}`, facultyPhysics.jwt);

    expect(response.status).toBe(200);
    expect(response.data.class?.subject).toBe('Physics');
  });

  it('Faculty Physics should NOT access Math department classes', async () => {
    const response = await makeAuthenticatedRequest('GET', `/api/classes/${mathClass.id}`, facultyPhysics.jwt);

    expect([403, 404, 200]).toContain(response.status);
    if (response.status === 200) {
      expect(response.data.class).toBeNull();
    }
  });

  it('Department query respects scope', async () => {
    const response = await makeAuthenticatedRequest('GET', '/api/classes', hodPhysics.jwt);

    expect(response.status).toBe(200);
    const classes = response.data.classes || [];
    
    // Should only have Physics classes
    const allPhysics = classes.every((c: any) => c.subject === 'Physics');
    expect(allPhysics).toBe(true);
  });
});
